pipeline {
    agent any
    stages {
		
		
		// ******ALL CODE TO BE ADDED BELOW THIS COMMENT******
		
		//Code starts for stage Clean
        stage('Clean') {
            steps {
                sh 'mvn clean'
            }
        }
		//Code ends for stage Clean
		
		
		//Code starts for stage PMD -  análisis estático de la calidad del código fuente en Java "básico" con PMD
		stage('PMD') {
            steps {
                sh 'mvn site'
            }
        }
		//Code ends for stage PMD
		
		
		//Code starts for stage Compile - compilar el código
		stage('Compile') {
            steps {
                sh 'mvn compile'
            }
        }
		//Code ends for stage Compile

		
		//Code starts for stage Code Analysis - análisis estático de la calidad del código fuente "avanzado" con SonarQube
		stage('Code Analysis') {
            steps {
			 script {
          scannerHome = tool 'sonar-scanner'
        }
				withSonarQubeEnv('My SonarQube Server')
				{
				sh "${scannerHome}/bin/sonar-scanner"
				}
			}
		}
		//Code ends for stage Code Analysis
		
		
		//Code starts for stage Unit Test - ejecución de pruebas unitarias con JUnit
        stage('Unit Test') {
            steps {
                sh 'mvn test'
            }	
            post {
                always {
                    junit 'target/surefire-reports/*.xml'
                }
            }
        }
		//Code ends for stage Unit Test
		
		
		//Code starts for stage Code Coverage - Cobertura de las pruebas unitarias anteriores, en este caso, para Java (todos los que dicen @test)
		stage('Code Coverage') {
            steps {
                sh 'mvn jacoco:report'
            }
        }
		//Code ends for stage Code Coverage

		
		// ******ALL CODE TO BE ADDED ABOVE THIS COMMENT******
    }
}
